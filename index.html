import 'dart:io';

void main() {
  var nomes = [];
  var paises = ['Brasil', 'Argentina', 'Colômbia', 'Peru'];
  List cidades = ['São Paulo', 'Buenos Aires', 'Bogotá', 'Lima'];
  List notas = [9.5, 8.7, 7.8, 6.9];
  List idades = [25, 30, 22, 28];
  List respostas = [true, false, true, false];
  var valores = [7.5, 'Ana', 10, 'Carlos'];
  stdout.writeln('Nomes: ${nomes.length}');
  stdout.writeln('Países: ${paises.length}');
  stdout.writeln('Cidades: ${cidades.length}');
  stdout.writeln('Notas: ${notas.length}');
  stdout.writeln('Idades: ${idades.length}');
  stdout.writeln('Respostas: ${respostas.length}');
  stdout.writeln('Valores: ${valores.length}');
}
		  
		  

// exemploLista2.dart

void main() {
  List frutas = [];
  print('A lista esta vazia? ${frutas.isEmpty}'); // true
  frutas = ['Maçã', 'Banana', 'Laranja', 'Uva'];
  print("Quantas frutas? ${frutas.length}");
  print('$frutas'); // 'Maçã', 'Banana', 'Laranja', 'Uva'
  print('${frutas[2]}'); // 'Laranja'
  print('${frutas[1]}'); // 'Banana'
  frutas[2] = 'Manga';
  frutas[1] = 'Abacaxi';
  print('$frutas'); // 'Maçã', 'Abacaxi', 'Manga', 'Uva'
  frutas[0] = 'Melancia';
  frutas[2] = 'Pera';
  print('$frutas'); // 'Melancia', 'Abacaxi', 'Pera', 'Uva'
  print('A lista esta vazia? ${frutas.isEmpty}'); //false
}		  
		  

// exemploLista3.dart

void main() {
  List nomes = ['Maria', 'João', 'Ana', 'Pedro'];
  // Usando índices inteiros
  print('Primeiro elemento: ${nomes[0]}');
  print('Primeiro elemento: ${nomes[nomes.length - 1]}');
  // Usando funções da lista
  print('Primeiro elemento: ${nomes.first}');
  print('Primeiro elemento: ${nomes.last}');
}		  
		  

// exemploLista4.dart

void main() {
  List nomes = ['Maria', 'João', 'Ana', 'Pedro', 'José'];

  for (int i = 0; i < nomes.length; i++) {
    print('Elemento da posição ${i + 1}: ${nomes[i]}');
  }
}		  
		  

// exemploLista5.dart

void main() {
  List nomes = ['Maria', 'João', 'Ana', 'Pedro', 'José'];

  for (var elemento in nomes) {
    print('Nome: $elemento');
    print('Posição (índice): ${nomes.indexOf(elemento)}');
  }
}
